example 
request
import requests
import json

url = "https://api.rfms.online/v2/payables"

payload = json.dumps([
  {
    "supplierName": "ACME",
    "invoiceNumber": "XS101",
    "invoiceDate": "9/24/19",
    "dueDate": "10/24/19",
    "discountableAmount": 0,
    "nonDiscountableAmount": 90,
    "discountRate": 0.1,
    "linkedDocumentNumber": "",
    "internalNotes ": "Internal note",
    "remittanceAdviceNotes": "Remit note",
    "detailLines": [
      {
        "storeNumber": 1,
        "accountCode": "1630",
        "subAccountCode": "",
        "amount": 100,
        "comment": "detail line"
      }
    ]
  }
])
headers = {
  'Content-Type': 'application/json'
}

response = requests.request("POST", url, headers=headers, data=payload)

print(response.text)
